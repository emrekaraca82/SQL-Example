1)ARAC MARKALARINA GÖRE SAYI GÖRE LÝSTELÝYÝNÝZ?
SELECT BRAND ,COUNT(BRAND) AS COUNT_ FROM WEBOFFERS WB INNER JOIN USER_ U
ON WB.USERID = U.ID GROUP BY BRAND ORDER BY 1

2)ARAÇ MARKASINA GÖRE HER MARKADAN KAÇ TANA OLDUPUNU YÜZDE GÖRE HESAPLAMA?
SELECT BRAND,COUNT(*) AS COUNT_ ,
ROUND(
CONVERT(FLOAT,COUNT(*))/(SELECT COUNT(*) FROM WEBOFFERS)*100,2
) AS PERCENT_
FROM WEBOFFERS
GROUP BY BRAND
ORDER BY 1
